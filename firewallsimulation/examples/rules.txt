# Sample rules
# Log all inbound SSH attempts from 10.0.0.0/8
action=LOG dir=IN proto=TCP dport=22 src=10.0.0.0/8

# Allow SSH inbound
action=ALLOW dir=IN proto=TCP dport=22

# Allow outbound DNS
action=ALLOW dir=OUT proto=UDP dport=53

# Block everything else inbound
action=BLOCK dir=IN
